//start
log in 
    insert flashdrive 
    scan flashdrive
    if flashdrive walang .csv file yung student or librarian
        ask if student or librarian 
            then forward to create account
    else 
        log in
    
if student
    while != exit
        show menu for student 
            borrow
                display books (sorted)
                choose book
                the book chosen will disappear
                note the book, student, time to the logs
                save the book to the student's usb

            return
                scan student's usb for the books na hiniram
                ask student if he/she wants to return the books 
                if yes
                    buburahin yung book sa usb ni student tapos lalagay sa database ng nasa laptop
                else   
                    cancel then return to menu
            exit

if librarian
    while != exit
        show books
            edit book
                verification
                edit yung name or date kung sakaling may typographical error sa mga books
            remove book
                verification
                if outdated na yung book may privilage si librarian na mag remove ng book  
            add book
                verification   
                if there are new books
            exit
